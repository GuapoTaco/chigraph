34 NodeInstance* GraphFunction::entryNode() const noexcept {
35  auto matching = nodesWithType("lang", "entry");
36 
37  if (matching.size() == 1) {
38  auto& vec = matching[0]->type().dataOutputs();
39  // make sure it has the same signature as the method
40  if (!std::equal(dataInputs().begin(), dataInputs().end(), vec.begin(), vec.end())) {
41  return nullptr;
42  }
43  // make sure it has the same exec names and size
44  auto foundExecOutputs = matching[0]->type().execOutputs();
45  if (!std::equal(execInputs().begin(), execInputs().end(), foundExecOutputs.begin(),
46  foundExecOutputs.end())) {
47  return nullptr;
48  }
49  return matching[0];
50  }
51  return nullptr;
52 }

98 Result GraphFunction::removeNode(NodeInstance* nodeToRemove) {
99  Result res;
100 
101  // disconnect it's connections
102 
103  // disconnect input exec
104  for (const auto& execSlot : nodeToRemove->inputExecConnections) {
105  for (const auto& pair : execSlot) {
106  if (pair.first != nullptr) { res += disconnectExec(*pair.first, pair.second); }
107  }
108  }
109  // disconnect output exec
110  auto ID = 0ull;
111  for (const auto& pair : nodeToRemove->outputExecConnections) {
112  if (pair.first != nullptr) { res += disconnectExec(*nodeToRemove, ID); }
113  ++ID;
114  }
115 
116  // disconnect input data
117  for (const auto& pair : nodeToRemove->inputDataConnections) {
118  if (pair.first != nullptr) {
119  res += disconnectData(*pair.first, pair.second, *nodeToRemove);
120  }
121  }
122 
123  // disconnect output data
124  ID = 0ull;
125  for (const auto& dataSlot : nodeToRemove->outputDataConnections) {
126  for (const auto& pair : dataSlot) {
127  if (pair.first != nullptr) { disconnectData(*nodeToRemove, ID, *pair.first); }
128  }
129  ++ID;
130  }
131  // then delete the node
132  nodes().erase(nodeToRemove->id());
133 
134  return res;
135 }

344 bool GraphFunction::removeLocalVariable(boost::string_view name) {
345  auto iter = std::find_if(mLocalVariables.begin(), mLocalVariables.end(),
346  [&](auto& toTest) { return toTest.name == name; });
347 
348  bool erased = false;
349  if (iter != mLocalVariables.end()) {
350  mLocalVariables.erase(iter);
351  erased = true;
352  }
353  if (!erased) { return false; }
354 
355  // remove set and get nodes
356  auto setNodes = nodesWithType(module().fullName(), "_set_" + name.to_string());
357  for (const auto& node : setNodes) { removeNode(node); }
358  auto getNodes = nodesWithType(module().fullName(), "_get_" + name.to_string());
359  for (const auto& node : getNodes) { removeNode(node); }
360 
361  return true;
362 }
