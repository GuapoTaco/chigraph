61  if (param.first->type().pure()) {
62  auto impure = node->type().pure() ? lastImpure : node;
63  const auto& cache = data.nodeCache[param.first];
64  if (cache.lastNodeCodegenned != impure) {
65  pureDependencies.push_back(param.first);
66  pureBBs.push_back(llvm::BasicBlock::Create(
67  node->context().llvmContext(),
68  boost::uuids::to_string(param.first->id()) + "____" +
69  boost::uuids::to_string(impure->id()),
70  f));
71  }
72  }
73  }
74  pureBBs.push_back(codeBlock);
75 
76  for (auto idx = 0ull; idx < pureDependencies.size(); ++idx) {
77  auto pureNode = pureDependencies[idx];
78  auto pureCodegenInto = pureBBs[idx];
79  auto pureTerminatorBlock = pureBBs[idx + 1];
80 
81  auto impure = node->type().pure() ? lastImpure : node;
82  codegenNode(pureNode, pureTerminatorBlock, impure, execInputID, pureCodegenInto,
83  data, res);
84  }
85  if (pureDependencies.empty()) {
86  pureBuilder.CreateBr(codeBlock);
87  } else {
88  pureBuilder.CreateBr(pureBBs[0]);
89  }
90  }
91 
92  size_t inputID = 0;
93  for (auto& param : node->inputDataConnections) {
94  // make sure everything is A-OK
95  if (param.first == nullptr) {
96  res.addEntry("EUKN", "No data input to node",
97  {{"nodeid", node->id()},
98  {"input ID", inputID},
99  {"nodetype", node->type().qualifiedName()}});
100 
101  return {boost::dynamic_bitset<>{}, std::vector<llvm::BasicBlock*>{}};
102  }
103 
104  auto cacheiter = data.nodeCache.find(param.first);
105  if (cacheiter == data.nodeCache.end()) {
106  res.addEntry("EUKN", "Failed to find in cache", {{"nodeid", param.first->id()}});
107 
108  return {boost::dynamic_bitset<>{}, std::vector<llvm::BasicBlock*>{}};
109  }
110 
111  auto& cacheObject = cacheiter->second;
112  if (param.second >= cacheObject.outputs.size()) {
113  res.addEntry("EUKN", "No data input to node",
114  {{"nodeid", node->id()},
115  {"input ID", inputID},
116  {"nodetype", node->type().qualifiedName()}});
117 
118  return {boost::dynamic_bitset<>{}, std::vector<llvm::BasicBlock*>{}};
119  }
120 
121  // get pointers to the objects
122  auto value = cacheObject.outputs[param.second];
123  // dereference
124  io.push_back(builder.CreateLoad(value));
125 
126  // make sure it's the right type
127  if (io[io.size() - 1]->getType() !=
128  node->type().dataInputs()[inputID].type.llvmType()) {
129  res.addEntry(
130  "EINT", "Internal codegen error: unexpected type in cache.",
131  {{"Expected LLVM type",
132  stringifyLLVMType(node->type().dataInputs()[inputID].type.llvmType())},
133  {"Found type", stringifyLLVMType(io[io.size() - 1]->getType())},
134  {"Node ID", node->id()},
135  {"Input ID", inputID}});
136  }
137  }
138  }
139 
140  // get outputs
141  {
142  llvm::IRBuilder<> allocBuilder(data.allocBlock);
143 
144  // create outputs
145  auto& outputCache = data.nodeCache[node].outputs;
146  outputCache.resize(node->type().dataOutputs().size(), nullptr);
147 
148  auto id = 0ull;
149  for (auto& output : node->type().dataOutputs()) {
150  if (outputCache[id] != nullptr) {
151  io.push_back(outputCache[id]);
152  ++id;
153  continue;
154  }
155 
156  llvm::AllocaInst* alloc = allocBuilder.CreateAlloca(
157  output.type.llvmType(), nullptr,
158  boost::uuids::to_string(node->id()) + "__" + std::to_string(id));
159  outputCache[id] = alloc;
160  io.push_back(alloc);
161 
162  // create debug info
163  {
164  // get type
165  llvm::DIType* dType = output.type.debugType();
166 
167  // TODO(#63): better names
168  auto debugVar = data.dbuilder->
169 #if LLVM_VERSION_MAJOR <= 3 && LLVM_VERSION_MINOR <= 7
170  createLocalVariable(
171  llvm::dwarf::DW_TAG_auto_variable,
172 #else
173  createAutoVariable(
174 #endif
175  data.diFunc,
176  boost::uuids::to_string(node->id()) + "__" + std::to_string(id),
177  data.diFunc->getFile(), 1, dType);
178 
179  data.dbuilder->insertDeclare(alloc, debugVar, data.dbuilder->createExpression(),
180  llvm::DebugLoc::get(1, 1, data.diFunc),
181  data.allocBlock);
182  }
183 
184  // make sure the type is right
185  if (llvm::PointerType::get(output.type.llvmType(), 0) != alloc->getType()) {
186  res.addEntry(
187  "EINT", "Internal codegen error: unexpected type returned from alloca.",
188  {{"Expected LLVM type",
189  stringifyLLVMType(llvm::PointerType::get(output.type.llvmType(), 0))},
190  {"Yielded type", stringifyLLVMType(alloc->getType())},
191  {"Node ID", node->id()}});
192  }
193  ++id;
194  }
195  }
196 
197  // add this block to the cache
198  if (node->type().qualifiedName() != "lang:entry") {
199  data.nodeCache[node].inputBlock.resize(
200  node->type().pure() ? 1 : node->inputExecConnections.size(), nullptr);
201  data.nodeCache[node].inputBlock[execInputID] = block;
202  }
203 
204  // create output blocks
205  std::vector<llvm::BasicBlock*> outputBlocks;
206  std::vector<llvm::BasicBlock*> unusedBlocks;
207  boost::dynamic_bitset<>
208  needsCodegen; // this holds if each block needs to recurse of it it's already in the cache
209  for (auto idx = 0ull; idx < node->outputExecConnections.size(); ++idx) {
210  // see if we can find it in the cache
211  NodeInstance* remoteNode;
212  size_t remotePortIdx;
213  std::tie(remoteNode, remotePortIdx) = node->outputExecConnections[idx];
214 
215  auto remoteCache =
216  data.nodeCache[remoteNode]; // this will construct it if it doesn't exist
217  if (remoteCache.inputBlock.size() > remotePortIdx &&
218  remoteCache.inputBlock[remotePortIdx] != nullptr) {
219  // this means it's already in the cache
220  outputBlocks.push_back(remoteCache.inputBlock[remotePortIdx]);
221  needsCodegen.push_back(false);
222  } else {
223  auto outBlock =
224  llvm::BasicBlock::Create(f->getContext(), node->type().execOutputs()[idx], f);
225  outputBlocks.push_back(outBlock);
226  needsCodegen.push_back(true); // these need codegen
227  if (node->outputExecConnections[idx].first != nullptr) {
228  outBlock->setName("node_" + boost::uuids::to_string(
229  node->outputExecConnections[idx].first->id()));
230  } else {
231  unusedBlocks.push_back(outBlock);
232  }
233  }
234  }
235  if (node->type().pure()) { outputBlocks.push_back(pureTerminator); }
236 
237  // codegen
238  res += node->type().codegen(
239  execInputID, llvm::DebugLoc::get(data.nodeLocations.right.at(node), 1, data.diFunc), io,
240  codeBlock, outputBlocks, data.compileCache);
241  if (!res) { return {boost::dynamic_bitset<>{}, std::vector<llvm::BasicBlock*>{}}; }
242 
243  // TODO(#64): raise an error here instead
244  for (auto& bb : unusedBlocks) {
245  llvm::IRBuilder<> builder(bb);
246 
247  builder.CreateRet(
248  llvm::ConstantInt::get(llvm::IntegerType::getInt32Ty(mod->getContext()), 0, true));
249  }
250 
251  data.nodeCache[node].lastNodeCodegenned = lastImpure;
252 
253  return {needsCodegen, outputBlocks};
254 }

259 void codegenHelper(NodeInstance* node, unsigned execInputID, llvm::BasicBlock* block,
260  codegenMetadata& data, Result& res) {
261  boost::dynamic_bitset<> needsCodegen;
262  std::vector<llvm::BasicBlock*> outputBlocks;
263  std::tie(needsCodegen, outputBlocks) =
264  codegenNode(node, nullptr, nullptr, execInputID, block, data, res);
265 
266  if (!res) { return; }
267 
268  // recurse!
269  for (auto idx = 0ull; idx < node->outputExecConnections.size(); ++idx) {
270  auto& output = node->outputExecConnections[idx];
271  if (output.first != nullptr && needsCodegen[idx]) {
272  codegenHelper(output.first, output.second, outputBlocks[idx], data, res);
273  }
274  }
