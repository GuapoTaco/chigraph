242 Result Context::compileModule(ChigModule& mod, std::unique_ptr<llvm::Module>* toFill) {
243  Expects(toFill != nullptr);
244 
245  Result res;
246 
247  auto llmod = std::make_unique<llvm::Module>(mod.fullName(), llvmContext());
248 
249  // generate dependencies
250  for (const auto& depName : mod.dependencies()) {
251  std::unique_ptr<llvm::Module> compiledDep;
252  res += compileModule(depName, &compiledDep); // TODO(#62): detect circular dependencies
253 
254  if (!res) { return res; }
255 
256 // link it in
257 #if LLVM_VERSION_MAJOR <= 3 && LLVM_VERSION_MINOR <= 7
258  llvm::Linker::LinkModules(llmod.get(), compiledDep.get());
259 #else
260  llvm::Linker::linkModules(*llmod, std::move(compiledDep));
261 #endif
262  }
263 
264  res += mod.generateModule(*llmod);
265 
266  // set debug info version if it doesn't already have it
267  if (llmod->getModuleFlag("Debug Info Version") == nullptr) {
268  llmod->addModuleFlag(llvm::Module::Warning, "Debug Info Version",
269  llvm::DEBUG_METADATA_VERSION);
270  }
271 
272  // verify the created module
273  if (res) {
274  bool errored;
275  std::string err;
276  {
277  llvm::raw_string_ostream os(err);
278  errored = llvm::verifyModule(*llmod, &os);
279  }
280 
281  if (errored) {
282  std::string moduleStr;
283  {
284  llvm::raw_string_ostream printerStr{moduleStr};
285  llmod->print(printerStr, nullptr);
286  }
287 
288  res.addEntry("EINT", "Internal compiler error: Invalid module created",
289  {{"Error", err}, {"Full Name", mod.fullName()}, {"Module", moduleStr}});
290  }
291  }
292 
293  *toFill = std::move(llmod);
294 
295  return res;
296 }
